## Quick Setup

```ts
import { registerPlugin } from '@canard/schema-form';
import { plugin } from '@canard/schema-form-mui-plugin';

registerPlugin(plugin);
```

## Component Mapping

Components are selected automatically based on JSON Schema properties. Priority order matches the list below (top = highest priority).

| Component | type | format | formType | enum | Notes |
|---|---|---|---|---|---|
| `FormTypeInputBooleanSwitch` | `boolean` | вҖ” | `"switch"` | вҖ” | MUI Switch |
| `FormTypeInputStringCheckbox` | `array` | вҖ” | `"checkbox"` | items.enum | MUI Checkbox group; items must be string schema |
| `FormTypeInputStringSwitch` | `string` | вҖ” | `"switch"` | 2 values | Toggle between two enum values |
| `FormTypeInputUri` | `string` | `"uri"` or formType `"uri"` | вҖ” | вҖ” | URI input with protocol dropdown |
| `FormTypeInputMonth` | `string` | `"month"` | вҖ” | вҖ” | MUI X MonthPicker |
| `FormTypeInputDate` | `string` | `"date"` | вҖ” | вҖ” | MUI X DatePicker |
| `FormTypeInputTime` | `string` | `"time"` | вҖ” | вҖ” | MUI X TimePicker (HH:mm:00Z format) |
| `FormTypeInputRadioGroup` | `string` \| `number` \| `integer` | вҖ” | `"radio"` or `"radiogroup"` | required | MUI RadioGroup |
| `FormTypeInputStringEnum` | `string` | вҖ” | вҖ” | required | MUI Select dropdown |
| `FormTypeInputArray` | `array` | вҖ” | вҖ” | вҖ” | Dynamic list with add/remove |
| `FormTypeInputSlider` | `number` \| `integer` | вҖ” | `"slider"` | вҖ” | MUI Slider |
| `FormTypeInputTextarea` | `string` | `"textarea"` or formType `"textarea"` | вҖ” | вҖ” | MUI TextField multiline |
| `FormTypeInputString` | `string` | вҖ” | вҖ” | вҖ” | MUI TextField (password via `format:"password"`) |
| `FormTypeInputNumber` | `number` \| `integer` | вҖ” | вҖ” | вҖ” | MUI TextField type=number |
| `FormTypeInputBoolean` | `boolean` | вҖ” | вҖ” | вҖ” | MUI Checkbox with indeterminate |

## Schema Extension Properties

```ts
{
  formType?: 'switch' | 'radio' | 'radiogroup' | 'checkbox' | 'slider' | 'textarea' | 'uri',
  format?:   'date' | 'time' | 'month' | 'password' | 'uri' | 'textarea',
}
```

## Usage Example

```tsx
import { Form } from '@canard/schema-form';

const schema = {
  type: 'object',
  properties: {
    name:     { type: 'string' },
    age:      { type: 'integer' },
    active:   { type: 'boolean', formType: 'switch' },
    role:     { type: 'string', enum: ['admin', 'user', 'guest'] },
    birthday: { type: 'string', format: 'date' },
    bio:      { type: 'string', format: 'textarea' },
    score:    { type: 'number', formType: 'slider', minimum: 0, maximum: 100 },
  },
};

<Form jsonSchema={schema} onChange={setValue} />
```

## MUI Context

Pass MUI styling options through `<Form context={...}>`:

```ts
type MuiContext = {
  size?:      'small' | 'medium';
  variant?:   'outlined' | 'filled' | 'standard';
  fullWidth?: boolean;
};
```

```tsx
<Form
  jsonSchema={schema}
  context={{ size: 'small', variant: 'outlined', fullWidth: true }}
  onChange={setValue}
/>
```

## Date/Time Picker Setup

MUI X date pickers require a `LocalizationProvider`. Wrap your app:

```tsx
import { LocalizationProvider } from '@mui/x-date-pickers';
import { AdapterDayjs } from '@mui/x-date-pickers/AdapterDayjs';

<LocalizationProvider dateAdapter={AdapterDayjs}>
  <Form jsonSchema={schema} onChange={setValue} />
</LocalizationProvider>
```
